  4. Data types 
    
            int varName    // integer data type for non decimal numbers
            float varName  // floating point data type for decimal numbers (single-precision)
            double varName // double precision data type for decimal numbers (double-precision)
            char varName   // character data type for single characters
            bool varName   // boolean data type for true or false
            void varName   // void data type for functions that do not return a value (absence of type)
  5. Declarations 
    
            int x;                      // Declare x to be an integer (value undefined)
            int x=255;                  // Declare and initialize x to 255
            int a[10];                  // Array of 10 ints (a[0] through a[9])
            int a[]={0,1,2};            // Initialized array (or a[3]={0,1,2}; )
            int a[2][2]={{1,2},{4,5}};  // Array of array of ints (2d array)
            char s[]="hello";           // String (6 elements including '\0')
            int* p;                     // p is a pointer to (address of) int
            char* s="hello";            // s points to unnamed array containing "hello"
            int& r=x;                   // r is a reference to (alias of) int x

  20. Decision Making Statements 
    
            // if statement
            if(condition) {
              // This block of code will get executed if the condition is True
            }

            // if-else statement
            if(condition) {
              // This block of code will get executed if the condition is True
            }else{
              // This block of code will get executed if the condition is False
            }

            // if else-if statement
            if (condition_1) {
              // This block of code will get executed if the condition_1 is True
            }else if (condition_2){
              // This block of code will get executed if the condition_1 is False but condition_2 is True
            }else{
              // This block of code will get executed if all conditions are False
            }

  23. Loops

              // This block of code will get executed while the condition is True
            }

            // do-while loop
            do {
              // This block of code will get executed at least once

            // for loop
            for(int i = 0; i < n; i++) {
              // This block of code will get executed n times
            }
  24. Break statement 
    break; // break keyword inside the loop is used to terminate the loop
  25. Continue statement 
    continue; // continue keyword skips the rest of the current iteration of the loop and returns to the starting point of the loop

  33. File Reading 
    
            #include <iostream>
            #include <fstream>
            using namespace std; 
            int main() {
              ifstream file;
              file.open("fileName.txt"); // Open text file
              string line;
              while(getline(file, line)) {
                cout << line << endl;
              }
              file.close(); // Close the file
            }
  34. File Open Modes 
    
            fs.open("test.txt", std::fstream::in)     // Open file for reading

            fs.open("test.txt", std::fstream::out)    // Open file for writing

            fs.open("test.txt", std::fstream::binary) // Open file in binary mode

            fs.open("test.txt", std::fstream::app)    // Opens the file and appends all the outputs at the end

            fs.open("test.txt", std::fstream::ate)    // Opens the file and sets the cursor to the end of the file 

            fs.open("test.txt", std::fstream::trunc)  // Removes the data in the existing file

            fs.open("test.txt", std::fstream::nocreate) // Opens the file only if it already exists

            fs.open("test.txt", std::fstream::noreplace) // Opens the file only if it does not exist
